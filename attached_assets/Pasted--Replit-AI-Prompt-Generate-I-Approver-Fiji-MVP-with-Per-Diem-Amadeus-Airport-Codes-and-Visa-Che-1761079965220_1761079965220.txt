## Cloudflare Worker — `workers/perDiem.ts`

Handles calculation logic and returns computed per diem values.

```ts
export default {
  async fetch(req: Request): Promise<Response> {
    if (req.method !== 'POST') return new Response('Method Not Allowed', { status: 405 });
    const { country, city, startDate, endDate } = await req.json();
    const sd = new Date(startDate);
    const ed = new Date(endDate);
    const days = Math.max(1, Math.ceil((ed.getTime() - sd.getTime()) / 86400000) + 1);
    const mieFJD = 100; // placeholder rate
    const first = mieFJD * 0.75;
    const middleDays = Math.max(0, days - 2) * mieFJD;
    const last = days > 1 ? mieFJD * 0.75 : 0;
    const total = first + middleDays + last;
    return new Response(JSON.stringify({ totalFJD: total, days, mieFJD }), { headers: { 'content-type': 'application/json' }});
  }
}
```

## Cloudflare Worker — `workers/visaCheck.ts`

Simulates visa status response.

```ts
export default {
  async fetch(req: Request): Promise<Response> {
    const { passportCountry, destinationCountry } = await req.json();
    if (passportCountry === 'Fiji' && destinationCountry === 'Australia') {
      return new Response(JSON.stringify({ status: 'ACTION', message: 'Visa required for Australia.' }), { headers: { 'content-type': 'application/json' }});
    }
    return new Response(JSON.stringify({ status: 'OK', message: 'No visa required.' }), { headers: { 'content-type': 'application/json' }});
  }
}
```

Proxy for Amadeus autocomplete.

export default {
  async fetch(req: Request): Promise<Response> {
    const url = new URL(req.url);
    const q = url.searchParams.get('q');
    if (!q) return new Response(JSON.stringify([]), { headers: { 'content-type': 'application/json' }});
    return new Response(JSON.stringify([
      { code: 'NAN', city: 'Nadi', country: 'Fiji' },
      { code: 'SUV', city: 'Suva', country: 'Fiji' }
    ]), { headers: { 'content-type': 'application/json' }});
  }
}

## React Components

### `/src/components/PerDiemCalculator.tsx`

```tsx
import React, { useEffect, useState } from 'react';
export default function PerDiemCalculator({ country, city, startDate, endDate, onUpdate }:{ country:string; city:string; startDate:string; endDate:string; onUpdate:(data:any)=>void; }){
  const [loading, setLoading] = useState(false);
  const [total, setTotal] = useState(0);
  useEffect(()=>{
    const ready = country && startDate && endDate;
    if(!ready) return;
    setLoading(true);
    fetch('/api/perDiem',{method:'POST',headers:{'content-type':'application/json'},body:JSON.stringify({country,city,startDate,endDate})})
      .then(r=>r.json())
      .then(data=>{ setTotal(data.totalFJD); onUpdate(data); })
      .finally(()=>setLoading(false));
  },[country,city,startDate,endDate]);
  return (<div className="p-4 rounded-xl shadow bg-white space-y-2">
    <div className="text-sm font-medium">Per Diem Calculation</div>
    {loading? <p>Calculating...</p>:<p className="text-lg font-semibold">Total: FJD {total.toFixed(2)}</p>}
  </div>);
}
```

### `/src/components/VisaChecker.tsx`

```tsx
import React, { useEffect, useState } from 'react';
export default function VisaChecker({ passportCountry, destinationCountry }:{ passportCountry:string; destinationCountry:string; }){
  const [status, setStatus] = useState<any>(null);
  useEffect(()=>{
    if(!passportCountry||!destinationCountry) return;
    fetch('/api/visaCheck',{method:'POST',headers:{'content-type':'application/json'},body:JSON.stringify({passportCountry,destinationCountry})})
      .then(r=>r.json()).then(setStatus);
  },[passportCountry,destinationCountry]);
  if(!status) return null;
  const cls = status.status==='OK'?'bg-green-100 text-green-800':status.status==='WARNING'?'bg-yellow-100 text-yellow-800':'bg-red-100 text-red-800';
  return <div className={`rounded-xl px-3 py-2 text-sm font-medium ${cls}`}>Visa: {status.status} — {status.message}</div>;
}
```
